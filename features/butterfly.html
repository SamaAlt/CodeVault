<div class="garden">
    <div class="butterfly">
        <div class="left-wing"></div>
        <div class="body"></div>
        <div class="right-wing"></div>
    </div>
</div>
.garden {
    width: 100%;
    height: 300px;
    background: linear-gradient(to bottom, #a8e6cf, #dcedc1);
    position: relative;
    overflow: hidden;
    border-radius: 10px;
}

.butterfly {
    position: absolute;
    width: 40px;
    height: 40px;
    animation: flight 15s infinite ease-in-out;
}

.left-wing, .right-wing {
    position: absolute;
    width: 20px;
    height: 30px;
    background-color: #ffaaa5;
    border-radius: 50% 50% 50% 50%;
    animation: flap 0.2s infinite ease-in-out;
}

.left-wing {
    left: 0;
    transform-origin: right center;
}

.right-wing {
    right: 0;
    transform-origin: left center;
}

.body {
    position: absolute;
    left: 50%;
    transform: translateX(-50%);
    width: 4px;
    height: 30px;
    background-color: #ff8b94;
    border-radius: 2px;
}

@keyframes flap {
    0%, 100% {
        transform: rotateY(0deg);
    }
    50% {
        transform: rotateY(60deg);
    }
}

@keyframes flight {
    0% {
        transform: translate(0, 0);
    }
    25% {
        transform: translate(200px, 100px);
    }
    50% {
        transform: translate(400px, 50px);
    }
    75% {
        transform: translate(200px, 150px);
    }
    100% {
        transform: translate(0, 0);
    }
}
document.getElementById('restart-flight').addEventListener('click', () => {
    const butterfly = document.querySelector('.butterfly');
    butterfly.style.animation = 'none';
    void butterfly.offsetWidth; // Trigger reflow
    butterfly.style.animation = 'flight 15s infinite ease-in-out';
});

Usage Notes

  • Great for decorative animations
  • Adjust flight path by modifying keyframes
  • Change colors to match your theme
  • Combine with other animations for complex effects
